							void AMoverObject::Tick(float DeltaTime)
							{
								Super::Tick(DeltaTime);

								// update location
								FVector CurrentLocation = GetActorLocation();
								FVector AdditionVector = GetActorForwardVector() * Speed * DeltaTime;
								FVector NewLocation = CurrentLocation + AdditionVector;
								SetActorLocation(NewLocation);

								// update rotation
								AddActorWorldRotation(FRotator(0,90 * DeltaTime, 0));
							}
